int mod_vectorAdd::createExercise(int type, int difficulty, string* exerciseParams)
{
    int i=1;
    ostringstream exercise;
    string exer;
    while(i<=6)
    {
        if(difficulty==1)
            exercise<<"1";  //first easy fraction
        else if(difficulty==2)         
        {
            if(randomizer(0,3)==0)  //probability 1/3 of fraction
                exercise<<randomizer(1,9);
            else
            exercise<<"1";     
        }
        else if(difficulty==3)        //only fractions if difficulty level 3
            exercise<<randomizer(1,9);
        else return -2;
        exercise<<";";
        if(randomizer(0,3)==0)  //probability 1/3 to get a negativ sign
            exercise<<"-";
        exercise<<randomizer(0,9); //first digit
        i++;
        exercise<<";";
    }
    exer=exercise.str(); //convert to string
    *exerciseParams=exer;
    return 0;
}
